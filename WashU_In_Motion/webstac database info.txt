create table in_class(

    userID SMALLINT UNSIGNED AUTO_INCREMENT,

    time_of_day DECIMAL(4,2),

    building VARCHAR(30) not null,

    d_week enum('M', 'T', 'W', 'H', 'F') not null,

    num_students SMALLINT UNSIGNED,

    primary key (userID)

)engine = InnoDB default character set = utf8 collate = utf8_general_ci;

create table out_of_class(

    userID SMALLINT UNSIGNED AUTO_INCREMENT,

    building VARCHAR(30) not null,

    d_week enum('M', 'T', 'W', 'H', 'F') not null,

    num_students SMALLINT UNSIGNED,

    time_of_day DECIMAL(4,2),

    primary key (userID)
